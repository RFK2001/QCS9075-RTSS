TEST_FUNC SpiTestStatusType_e TC_SPI_012( TestCaseType TCType, uint8 cpu_id )
{

	SpiTestStatusType_e ulTestStatus = SPI_PASS;
	uint8 ucCount = 0;
	Std_ReturnType Std_Return = E_OK;

	*(uint32*)spiSE4LOOPBACK_ADDR = 0x01;

	for (int i =0; i < MAX_LB_TRANS_LEN; i++)
	{
		InputBuffer[i] = i;
		OutputBuffer[i] =  0x0;
	}
	Std_Return |= Spi_SetupEB( SpiConf_SpiChannel_SpiChannel_18, InputBuffer, OutputBuffer, MAX_LB_TRANS_LEN );
	Std_Return = Spi_AsyncTransmit( SpiConf_SpiSequence_SpiSequence_14 );
	if( E_OK != Std_Return )
	{
		if(TCType == UARTFRAMEWORK)
		{
		    DEBUG_LOG(SAIL_INFO,"Spi_AsyncTransmit(SpiConf_SpiSequence_SpiSequence_14): failed!\r\n" );
		}
	}
	else
	{
		if( pdPASS != ( xEventGroupWaitBits(  xEventGroupUssHandle, LOOPBACK_SEQ14_CBKWAIT_MSK, pdTRUE, pdFALSE , &xEvBitsSet, safertosapiMAX_DELAY ) ) )
		{
			if(TCType == UARTFRAMEWORK)
			{
			    DEBUG_LOG(SAIL_INFO,"xEventGroupWaitBits(LOOPBACK_SEQ14_CBKWAIT_MSK) failed!\r\n");
			}
		}
		else
		{
			for(ucCount = 0; ucCount < MAX_LB_TRANS_LEN; ucCount++)
			{
				if( OutputBuffer[ucCount] != InputBuffer[ucCount] )
				{
					ulTestStatus = SPI_FAIL;
					break;
				}
			}
		}
	}
    if(TCType == UARTFRAMEWORK)
    {
	    prvSPITestDebugPrint( ulTestStatus, "Test - SE4 Loopback ", 12 );
    }
    else if(TCType ==DVFRAMEWORK)
    {
    	if(ulTestStatus == SPI_FAIL)
    	{
    		DebugPrint_DVTask(DVTASK_FAILED,1);
    	}
    }

	*(uint32*)spiSE4LOOPBACK_ADDR = 0x0;
	return ulTestStatus;
}
